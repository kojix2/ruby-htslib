def each
  return to_enum(__method__) unless block_given?

  @n_cigar.times do |i|
    c = @pointer[i].read_uint32
    op =  LibHTS.bam_cigar_opchr(c)
    len = LibHTS.bam_cigar_oplen(c)
    yield [op, len]
  end
end
