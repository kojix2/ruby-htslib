def chrom_size(chrom)
  unless chrom.is_a?(String) || chrom.is_a?(Symbol)
    # FIXME
    raise ArgumentError, "Expect chrom to be String or Symbol"
  end

  chrom = chrom.to_s
  result = LibHTS.faidx_seq_len(@fai, chrom)
  result == -1 ? nil : result
end
